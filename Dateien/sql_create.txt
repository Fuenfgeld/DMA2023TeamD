DROP TABLE IF EXISTS patients; 

DROP TABLE IF EXISTS conditions; 

DROP TABLE IF EXISTS medications; 

DROP TABLE IF EXISTS encounters;

CREATE TABLE IF NOT EXISTS patients (
                           Id STRING,
                           BIRTHDATE DATE,
                           DEATHDATE DATE,
                           SSN STRING,
                           DRIVERS STRING,
                           PASSPORT STRING,
                           PREFIX STRING,
                           FIRST STRING,
                           LAST STRING,
                           SUFFIX STRING,
                           MAIDEN STRING,
                           MARITAL STRING,
                           RACE STRING,
                           ETHNICITY STRING,
                           GENDER STRING,
                           BIRTHPLACE STRING,
                           ADDRESS STRING,
                           CITY STRING,
                           STATE STRING,
                           COUNTY STRING,
                           ZIP STRING,
                           LAT INTEGER,
                           LON INTEGER,
                           HEALTHCARE_EXPENSES INTEGER,
                           HEALTHCARE_COVERAGE INTEGER,
                           PRIMARY KEY (Id)
                       );

CREATE TABLE IF NOT EXISTS conditions (
                           START DATE,
                           STOP DATE,
                           PATIENT STRING,
                           ENCOUNTER STRING,
                           CODE STRING,
                           DESCRIPTION STRING,
                           FOREIGN KEY (PATIENT)
                              REFERENCES patients (Id)
                       );

CREATE TABLE IF NOT EXISTS medications (
                           START DATE,
                           STOP DATE,
                           PATIENT STRING,
                           PAYER STRING,
                           ENCOUNTER STRING,
                           CODE STRING,
                           DESCRIPTION STRING,
                           BASE_COST INTEGER,
                           PAYER_COVERAGE INTEGER,
                           DISPENSES INTEGER,
                           TOTALCOST INTEGER,
                           REASONCODE STRING,
                           REASONDESCRIPTION STRING,
                           FOREIGN KEY (PATIENT)
                              REFERENCES patients (Id)   
                       );

CREATE TABLE IF NOT EXISTS encounters (
			   Id STRING,
			   START DATE,
                           STOP DATE,
                           PATIENT STRING,
                           ORGANIZATION	STRING,
                           PROVIDER STRING,
                           DESCRIPTION STRING,
			   PAYER STRING,
			   ENCOUNTERCLASS STRING,
			   CODE	STRING,
			   DESCRIPTION STRING,
                           BASE_COST INTEGER,
			   TOTAL_CLAIM_COST INTEGER,
			   PAYER_COVERAGE INTEGER,
                           REASONCODE STRING,
                           REASONDESCRIPTION STRING,
                           FOREIGN KEY (PATIENT)
                              REFERENCES patients (Id) 
                       );
